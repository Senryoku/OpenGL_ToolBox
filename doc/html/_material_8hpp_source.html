169  template<typename T>
170  void setAttribute(typename std::vector<Uniform<T>>& Container, const std::string& Name, T Value)
171  {
172  GLuint Location = glGetUniformLocation(_shadingProgram->getID(), Name.c_str());
173 
174  if(Location >= 0)
175  {
176  for(Uniform<T>& U : Container)
177  {
178  if(U.getName() == Name)
179  {
180  U.setValue(Value);
181  return;
182  }
183  }
184  Container.push_back(Uniform<T>{Name, Location, Value});
185  } else
186  std::cerr << "Material: Uniform " + Name + " not found." << std::endl;
187  }
